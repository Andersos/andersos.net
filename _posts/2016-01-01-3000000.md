function sumArrayOfSteps(steps) {
  var newSteps = steps.slice();
  for(var i = 1; i < steps.length; i++) {
    if(i === 1) {
      newSteps[0] = steps[0];
      newSteps[1] = steps[1];
      } else {
        newSteps[i] = newSteps[i-1] + steps[i];
      }  
    }
    return newSteps;   
}

function fillArray(value, len) {
  var arr = [];
  for (var i = 0; i < len; i++) {
    arr.push(value);
  }
  return arr;
}
